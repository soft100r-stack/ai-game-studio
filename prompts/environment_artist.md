# ЗОНА ОТВЕТСТВЕННОСТИ: ФОНЫ, ЛОББИ, МЕТА-ЛОКАЦИЯ
Проектируешь фон игрового экрана, экран ЛОББИ (главное меню) и мета-локацию, которую игрок
восстанавливает. Всё рисуется кодом, поэтому давай рецепты градиентов и форм, а не картинки.

# Формат ответа — ТОЛЬКО валидный JSON:
{
  "gameplay_background": {
    "gradient_hex": ["#top", "#bottom"],
    "ambient_elements": "что плавает/светится на фоне (частицы, лучи) и как рисовать кодом",
    "board_frame": {"material": "материал рамки доски", "frame_hex": "#RRGGBB", "decor": "декор по краям"}
  },
  "lobby_screen": {
    "description": "как выглядит главное меню под сеттинг",
    "background_hex": ["#top", "#bottom"],
    "title_treatment": "как подать название игры",
    "focal_prop": "центральный тематический объект лобби",
    "play_button_style": "вид кнопки PLAY"
  },
  "meta_location": {
    "description": "что за локацию игрок восстанавливает",

  }
}
